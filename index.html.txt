<script>
    // --- GANTI URL INI DENGAN URL DEPLOYMENT ANDA ---
    const SCRIPT_URL = "URL_GOOGLE_SCRIPT_ANDA_DISINI"; 
    // ------------------------------------------------

    // 1. Dark Mode Logic
    function toggleTheme() {
        const body = document.body;
        const currentTheme = body.getAttribute("data-theme");
        const newTheme = currentTheme === "dark" ? "light" : "dark";
        body.setAttribute("data-theme", newTheme);
        localStorage.setItem("theme", newTheme);
    }
    
    // Load saved theme
    if(localStorage.getItem("theme") === "dark") {
        document.body.setAttribute("data-theme", "dark");
    }

    // 2. Custom Condition Logic (Show/Hide Text Input)
    function checkCustom(selectElement) {
        const customInput = selectElement.nextElementSibling;
        if (selectElement.value === "Lainnya") {
            customInput.classList.remove("hidden");
            customInput.required = true;
        } else {
            customInput.classList.add("hidden");
            customInput.required = false;
        }
    }

    // 3. Add Multiple Items Logic
    let itemCount = 1;
    function addItem() {
        const container = document.getElementById("itemsContainer");
        const newItem = document.createElement("div");
        newItem.classList.add("item-card");
        newItem.id = `item-${itemCount}`;
        newItem.innerHTML = `
            <button type="button" class="btn-remove" onclick="removeItem('${newItem.id}')">Hapus</button>
            <label>Nama Barang</label>
            <input type="text" class="nama-barang" required placeholder="Nama item">
            <div style="display:flex; gap:10px;">
                <div style="flex:1;">
                    <label>Jumlah</label>
                    <input type="number" class="jumlah" min="1" value="1" required>
                </div>
            </div>
            <label>Kondisi Awal</label>
            <select class="kondisi-awal" onchange="checkCustom(this)">
                <option value="Baik">Baik</option>
                <option value="Rusak">Rusak</option>
                <option value="Lainnya">Lainnya</option>
            </select>
            <input type="text" class="custom-input hidden" placeholder="Ketik kondisi awal...">
            <label>Kondisi Akhir</label>
            <select class="kondisi-akhir" onchange="checkCustom(this)">
                <option value="Baik">Baik</option>
                <option value="Rusak">Rusak</option>
                <option value="Lainnya">Lainnya</option>
            </select>
            <input type="text" class="custom-input hidden" placeholder="Ketik kondisi akhir...">
        `;
        container.appendChild(newItem);
        itemCount++;
    }

    function removeItem(id) {
        document.getElementById(id).remove();
    }

    // 4. Form Submission Logic
    document.getElementById('inventoryForm').addEventListener('submit', function(e) {
        e.preventDefault();
        
        if(SCRIPT_URL === "https://script.google.com/macros/s/AKfycbzpbAdwEnzKMI5MwvXgf03ihUgDOGHq5Nwwog14KzQwTq1DQPLpyv9qtbgQWM6u1LQoOA/exec") {
            alert("Error: Anda belum memasukkan URL Google Script ke dalam kode HTML.");
            return;
        }

        document.getElementById('loadingOverlay').style.display = 'flex';

        // Collect Main Data
        const formData = {
            peminjam: document.getElementById('nama').value,
            departemen: document.getElementById('dept').value,
            tglPinjam: document.getElementById('tglPinjam').value,
            tglKembali: document.getElementById('tglKembali').value,
            items: []
        };

        // Collect Items Data
        const itemCards = document.querySelectorAll('.item-card');
        itemCards.forEach(card => {
            const namaBarang = card.querySelector('.nama-barang').value;
            const jumlah = card.querySelector('.jumlah').value;
            
            // Handle Custom Select Logic
            const selAwal = card.querySelector('.kondisi-awal');
            const valAwal = selAwal.value === "Lainnya" ? selAwal.nextElementSibling.value : selAwal.value;
            
            const selAkhir = card.querySelector('.kondisi-akhir');
            const valAkhir = selAkhir.value === "Lainnya" ? selAkhir.nextElementSibling.value : selAkhir.value;

            formData.items.push({
                namaBarang: namaBarang,
                jumlah: jumlah,
                kondisiAwal: valAwal,
                kondisiAkhir: valAkhir
            });
        });

        // Send to Google Script
        fetch(SCRIPT_URL, {
            method: "POST",
            body: JSON.stringify(formData)
        })
        .then(response => response.json())
        .then(data => {
            document.getElementById('loadingOverlay').style.display = 'none';
            if(data.result === "success") {
                alert("Data Berhasil Disimpan!");
                document.getElementById('inventoryForm').reset();
                // Reset item container to 1 item
                document.getElementById("itemsContainer").innerHTML = "";
                itemCount = 0;
                addItem(); 
            } else {
                alert("Gagal menyimpan data: " + data.error);
            }
        })
        .catch(error => {
            document.getElementById('loadingOverlay').style.display = 'none';
            alert("Terjadi kesalahan jaringan.");
            console.error('Error:', error);
        });
    });
</script>
